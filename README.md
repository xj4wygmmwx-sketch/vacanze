  cal.appendChild(empty);

for(let d=1; d<=31; d++){
  const el = document.createElement('div');
  el.className='day';
  const dow = (firstDow + d - 1) % 7;
  const isWeekend = dow===5 || dow===6;
  if(isWeekend) el.classList.add('weekend');

  const stageIds = dayMap[d] || [];
  const primaryStage = stageIds.length ? stops.find(s=>s.id===stageIds[stageIds.length-1]) : null;
  if(primaryStage){
    el.classList.add('active-stage');
    el.style.setProperty('--stage-color', primaryStage.hex);
  }
  el.innerHTML = `${d}${travelDays.has(d)?'<span class="travel-mark">🚗</span>':''}`;
  el.addEventListener('click', ()=>{
    document.querySelectorAll('.day.selected').forEach(x=>x.classList.remove('selected'));
    el.classList.add('selected');
    if(travelDays.has(d)){
      const leg = legs.find(l=>parseInt(l.date)===d);
      showLeg(leg);
      resetView();
    } else if(primaryStage){
      showStop(primaryStage.id);
      focusStop(primaryStage);
    } else {
      panel.innerHTML = `<div class="empty-state">Nessuna tappa in programma<br>per questo giorno.</div>`;
    }
  });
  cal.appendChild(el);
}

let checklistState = {};
try{
  checklistState = JSON.parse(localStorage.getItem('andalucia2026-checklist') || '{}');
}catch(e){ checklistState = {}; }

function saveChecklistState(){
  try{ localStorage.setItem('andalucia2026-checklist', JSON.stringify(checklistState)); }catch(e){}
}

function updateProgress(){
  const allItems = Object.values(checklistData).flat();
  const checkedCount = allItems.filter(i=>checklistState[i]).length;
  checklistProgress.textContent = `${checkedCount}/${allItems.length}`;
}

function renderChecklist(){
  checklistBody.innerHTML = '';
  Object.entries(checklistData).forEach(([cat, items])=>{
    const catEl = document.createElement('div');
    catEl.className = 'checklist-cat';
    const h4 = document.createElement('h4');
    h4.textContent = cat;
    catEl.appendChild(h4);
    items.forEach(item=>{
      const row = document.createElement('div');
      row.className = 'check-item' + (checklistState[item] ? ' checked' : '');
      row.innerHTML = `<span class="box">${checklistState[item] ? '✓' : ''}</span><span class="label">${item}</span>`;
      row.addEventListener('click', ()=>{
        checklistState[item] = !checklistState[item];
        row.classList.toggle('checked');
        row.querySelector('.box').textContent = checklistState[item] ? '✓' : '';
        saveChecklistState();
        updateProgress();
      });
      catEl.appendChild(row);
    });
    checklistBody.appendChild(catEl);
  });
  updateProgress();
}
renderChecklist();
